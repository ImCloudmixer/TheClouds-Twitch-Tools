Twitch Channel Scraper

Copy the JavaScript code below and run it in your browser's console on the Twitch page.

function getTwitchChannelNames() {
    const channels = document.querySelectorAll('[data-a-id^="card-"]');
    const channelNamesSet = new Set();
	
    // Change array amount, make sure to scroll down page. 
    Array.from(channels).slice(0, 600).forEach(el => {
        const channelName = el.getAttribute('data-a-id').slice(5);
        channelNamesSet.add(channelName);
    });
    const uniqueChannelNames = Array.from(channelNamesSet);
    console.log(uniqueChannelNames.join(', '));
    return uniqueChannelNames;
}
getTwitchChannelNames();
